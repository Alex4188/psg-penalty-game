<script>
let currentShot = 1;
let goals = 0;

function startGame() {
  currentShot = 1;
  goals = 0;
  document.getElementById("goals").textContent = goals;
  document.getElementById("shot").textContent = currentShot;
  show("game");
}

function kick(direction) {
  const keeperMove = ["left","center","right"][Math.floor(Math.random()*3)];
  const keeper = document.getElementById("keeper");
  const ball = document.getElementById("ball");

  if (keeperMove === "left") keeper.style.left = "30px";
  if (keeperMove === "center") keeper.style.left = "130px";
  if (keeperMove === "right") keeper.style.left = "230px";

  ball.style.opacity = 1;
  ball.style.left = direction === "left" ? "60px" :
                    direction === "right" ? "240px" : "150px";
  ball.style.bottom = "120px";

  setTimeout(() => {
    ball.style.opacity = 0;
    ball.style.bottom = "10px";

    if (direction !== keeperMove) {
      goals++;
      document.getElementById("goals").textContent = goals;
    }

    currentShot++;

    if (currentShot > 5) {
      document.getElementById("finalScore").textContent = goals;
      show("result");
    } else {
      document.getElementById("shot").textContent = currentShot;
    }
  }, 500);
}

function show(screen) {
  ["menu","game","result"].forEach(id =>
    document.getElementById(id).classList.remove("active")
  );
  document.getElementById(screen).classList.add("active");
}
